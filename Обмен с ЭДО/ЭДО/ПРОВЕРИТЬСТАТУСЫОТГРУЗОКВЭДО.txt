// По коду предприятия необходимо определить систему ЭДО, с которой работает пользователь.
// В зависимости от системы вызвать нужную функцию.

СИСТЕМАЭДО		= ЗАПРОС( "SELECT value FROM param_ex WHERE param= 'EDI_EDI" + ПРЕДПРИЯТИЕ + "'" );

IF ( СИСТЕМАЭДО == "СБИС" )
	RETURN ЭДО.СБИСПроверкаСтатусаОтгрузок( ПРЕДПРИЯТИЕ );
ELSE IF ( СИСТЕМАЭДО == "ТАКСКОМ" )
{
	ЭДО.ТАКСКОМОбработкаСлужебныхЭтапов( ПРЕДПРИЯТИЕ );
	ЭДО.ТАКСКОМПроверкаСтатусаПриходов( ПРЕДПРИЯТИЕ );
	RETURN ЭДО.ТАКСКОМПроверкаСтатусаОтгрузок( ПРЕДПРИЯТИЕ );
}

RETURN false;
