
СИСТЕМНОЕСООБЩЕНИЕ( "Пожалуйста, подождите. Обмен с ГИС Маркировка" );

// Для начала проверяем, может быть есть уже отправленные ранее документы с запросом GTIN
ТаблицаДокументов		= МАРКИРОВКА.ЗАПРОССПИСКАДОКУМЕНТОВ( ПРЕДПРИЯТИЕ, "OST_DESCRIPTION,OST_DESCRIPTION_CSV,OST_DESCRIPTION_XML" );
IF ( !ПУСТО( ТаблицаДокументов ) )
{
	ДОБАВИТЬКОНТЕКСТ( "SELECT document FROM " + ТаблицаДокументов, "СписокДокументов" );
	ПОКА ( !КОНЕЦКОНТЕКСТА( "СписокДокументов" ) )
	{
		МАРКИРОВКА.ОБРАБОТАТЬДОКУМЕНТGTIN( ПРЕДПРИЯТИЕ, СписокДокументов.document );
		ПРОПУСТИТЬ( 1, "СписокДокументов" );
	}
	УДАЛИТЬКОНТЕКСТ( "СписокДокументов" );
}
СИСТЕМНОЕСООБЩЕНИЕ( );
RETURN !ПУСТО( ТаблицаДокументов );
