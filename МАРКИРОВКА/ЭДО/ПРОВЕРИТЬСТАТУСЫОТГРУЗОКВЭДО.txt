// По коду предприятия необходимо определить систему ЭДО, с которой работает пользователь.// В зависимости от системы вызвать нужную функцию.СИСТЕМАЭДО		= ЗАПРОС( "SELECT value FROM param_ex WHERE param= 'MARK_EDI" + ПРЕДПРИЯТИЕ + "'" );IF ( СИСТЕМАЭДО == "ЭВОТОР" )	RETURN МАРКИРОВКА.ЭВОТОРПроверкаСтатусаОтгрузок( ПРЕДПРИЯТИЕ );ELSE IF ( СИСТЕМАЭДО == "КОНТУР" )	RETURN МАРКИРОВКА.КОНТУРПроверкаСтатусаОтгрузок( ПРЕДПРИЯТИЕ );ELSE IF ( СИСТЕМАЭДО == "СБИС" )	RETURN МАРКИРОВКА.СБИСПроверкаСтатусаОтгрузок( ПРЕДПРИЯТИЕ );ELSE IF ( СИСТЕМАЭДО == "ТАКСКОМ" ){	МАРКИРОВКА.ТАКСКОМОбработкаСлужебныхЭтапов( ПРЕДПРИЯТИЕ );	МАРКИРОВКА.ТАКСКОМПроверкаСтатусаПриходов( ПРЕДПРИЯТИЕ );	RETURN МАРКИРОВКА.ТАКСКОМПроверкаСтатусаОтгрузок( ПРЕДПРИЯТИЕ );}RETURN false;
