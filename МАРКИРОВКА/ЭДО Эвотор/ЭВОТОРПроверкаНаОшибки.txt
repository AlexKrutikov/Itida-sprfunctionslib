Результат		= true;IF ( _ERRORCODE !=  0 ) {	Ответ		= _ERRORDESCRIPTION + CHR( 13 ) + Ответ;	Результат	= false;}ELSE IF ( AT( Ответ, "error_description" ) > 0 ) {	ЗАГРУЗИТЬJSON( "Ошибка", ПЕРЕКОДИРОВАТЬ( Ответ, "UTF-8", "ANSI" ) );	Ответ		= "Ошибка: " + ЗНАЧЕНИЕПОЛЯ( "Ошибка", "error_description" );	Результат	= false;}ELSE IF ( AT( Ответ, "error" ) > 0 ) {	ЗАГРУЗИТЬJSON( "Ошибка", ПЕРЕКОДИРОВАТЬ( Ответ, "UTF-8", "ANSI" ) );	Ответ		= "Ошибка: " + ЗНАЧЕНИЕПОЛЯ( "Ошибка", "message" );	Результат	= false;}RETURN Результат;
