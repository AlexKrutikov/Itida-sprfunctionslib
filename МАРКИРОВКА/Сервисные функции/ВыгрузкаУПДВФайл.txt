// Запрашиваем имя файла для загрузкиИмяФайла		= Маркировка.СоздатьXMLИзДокумента( ПРЕДПРИЯТИЕ, GUIDДокумента );IF ( ПУСТО( ИмяФайла ) ) RETURN false;ИмяФайлаБезПути	= ИмяФайла;IF ( ATC( ИмяФайла, "\" ) > 0 )	ИмяФайлаБезПути	= ПОДСТРОКА( ИмяФайла, RAT( ИмяФайла, "\" ) + 1 );IF ( ТИП( "_МАРКИРОВКАПУТЬДЛЯСОХРАНЕНИЯФАЙЛОВ" ) == "U" ) 	_МАРКИРОВКАПУТЬДЛЯСОХРАНЕНИЯФАЙЛОВ	= "";	IF ( ПУСТО( _МАРКИРОВКАПУТЬДЛЯСОХРАНЕНИЯФАЙЛОВ ) ){	НовоеИмяФайла						= ФАЙЛВЫБРАТЬ( false, "Файлы XML|*.xml", 526348, "xml", ИмяФайла );	_МАРКИРОВКАПУТЬДЛЯСОХРАНЕНИЯФАЙЛОВ	= ЛЕВСИМВ( НовоеИмяФайла, RAT( НовоеИмяФайла, "\" ) );}IF ( ПРАВСИМВ( _МАРКИРОВКАПУТЬДЛЯСОХРАНЕНИЯФАЙЛОВ, 1 ) != "\" )	_МАРКИРОВКАПУТЬДЛЯСОХРАНЕНИЯФАЙЛОВ	+= "\";НовоеИмяФайла		= _МАРКИРОВКАПУТЬДЛЯСОХРАНЕНИЯФАЙЛОВ + ИмяФайлаБезПути; RETURN ФАЙЛПЕРЕИМЕНОВАТЬ( ИмяФайла, НовоеИмяФайла, 11 );
