// Запрашиваем имя файла для загрузкиИмяФайла		= ФАЙЛВЫБРАТЬ( 1, "Файлы XML|*.xml" );IF ( ПУСТО( ИмяФайла ) ) RETURN false;Файл					= ФАЙЛОТКРЫТЬ( ИмяФайла );IF ( Файл < 0 ) {	СООБЩЕНИЕ( "Не удалось открыть файл '" + ИмяФайла + "' для загрузки данных." );	RETURN "";}СодержимоеФайла			= ФАЙЛПРОЧИТАТЬ( Файл, "S", ФАЙЛРАЗМЕР( Файл ) );ФАЙЛЗАКРЫТЬ( Файл );IF ( ПУСТО( Маркировка.СоздатьДокументИЗXML( Предприятие, GUIDДокумента, "", СодержимоеФайла, true ) ) )	RETURN false;//-------- Переименовываем файл, чтобы не мешалФАЙЛПЕРЕИМЕНОВАТЬ( ИмяФайла, ИмяФайла + ".bak" );
