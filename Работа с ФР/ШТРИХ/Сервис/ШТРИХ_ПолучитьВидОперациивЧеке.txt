IF ( ТипЧека == 0 ) //приход
	ВидОперации = 1; 
ELSE IF ( ТипЧека == 1 ) //возврат прихода
	ВидОперации = 2;
ELSE IF ( ТипЧека == 4 ) //расход
	ВидОперации = 3;
ELSE IF ( ТипЧека == 5 ) //возврат расхода
	ВидОперации = 4;
ELSE
	ВидОперации = 1; //по умолчанию возвращаем Продажу (приход)
	
RETURN ВидОперации;
