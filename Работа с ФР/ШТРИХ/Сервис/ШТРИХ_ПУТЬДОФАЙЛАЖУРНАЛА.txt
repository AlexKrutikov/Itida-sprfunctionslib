ИмяФайлаЛога = "";

ПутьДоФайла = ПОКРУЖЕНИЯ("appdata");

IF (ПУСТО(ПутьДоФайла))
	ПутьДоФайла = "C:\Users\Public\Documents";
//на всякий случай проверим существование каталога
IF ( !ФАЙЛ(ПутьДоФайла, 1) )
{
	//если нет каталога в системе, то будем далее пытаться писать в каталог с Айтидой
	ПутьДоФайла = "";
}
ELSE
{
	//если каталог есть в системе, то проверим наличие вложенного каталога Itida
	IF ( !ФАЙЛ(ПутьДоФайла + "\Itida", 1) )
	{
		//если нет каталога Itida, то создадим его
		IF ( КАТАЛОГСОЗДАТЬ(ПутьДоФайла + "\Itida") )
		{
			//если каталог создался, то меняем полный путь с учетом созданного каталога
			ПутьДоФайла = ПутьДоФайла + "\Itida";
		}
	}
	ELSE
	{
		ПутьДоФайла = ПутьДоФайла + "\Itida";
	}
}
ИмяФайлаЛога = ПутьДоФайла + ЕСЛИ(ПУСТО(ПутьДоФайла),"","\") + "kkt_shtrih_log_" + КОДОБОРУДОВАНИЯ + ".txt";

RETURN ИмяФайлаЛога;
