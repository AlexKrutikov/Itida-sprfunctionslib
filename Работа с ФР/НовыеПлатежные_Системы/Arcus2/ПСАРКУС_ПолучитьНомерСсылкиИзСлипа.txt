ТребуемоеЗначение = "";

РазделительСтрок 	= CHR(10);
НомерСтрокиСлипа	= 1;
СлипЧекВрем			= РазделительСтрок + СЛИПЧЕК;
СтрокаСлипа			= ПЕРЕКОДИРОВАТЬ( ПОЛУЧИТЬСЛОВО( СлипЧекВрем, РазделительСтрок, НомерСтрокиСлипа ), "ANSI", "UTF-8" );

WHILE ( !ПУСТО( СтрокаСлипа ) )
{
	IF ( ATC( UPPER( СтрокаСлипа ), "НОМЕР ССЫЛКИ" ) > 0 )
	{
		ТребуемоеЗначение = ALLTRIM( STRTRANC( STRTRANC( СтрокаСлипа, "НОМЕР ССЫЛКИ", "" ), ":", "") );
		BREAK;
	}
		
	НомерСтрокиСлипа++;
	СтрокаСлипа		= ПЕРЕКОДИРОВАТЬ( ПОЛУЧИТЬСЛОВО( СлипЧекВрем, РазделительСтрок, НомерСтрокиСлипа ), "ANSI", "UTF-8" );
}

RETURN ТребуемоеЗначение;
