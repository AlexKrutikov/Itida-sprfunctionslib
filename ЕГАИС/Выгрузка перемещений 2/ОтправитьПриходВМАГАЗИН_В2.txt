);		УДАЛИТЬКОНТЕКСТ( "ДокументИнфо" );				СООБЩЕНИЕ( ТекстСообщения + CHR( 13 ) + "Документ не был отправлен в ЕГАИС" );		RETURN false;	}		ТекстНакладной	+= "					<tts:Position>						<tts:Identity>" + STR( НОМЕРСТРОКИ( "СписокСтрок" ) + 1 ) + "</tts:Identity>						<tts:ProductCode>" + СписокСтрок.alcocode + "</tts:ProductCode>						<tts:Quantity>" +
