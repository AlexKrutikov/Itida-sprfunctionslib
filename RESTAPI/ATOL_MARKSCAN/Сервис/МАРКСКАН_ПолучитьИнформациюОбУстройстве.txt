АдресТСД = RESTAPI.ПолучитьАдресТСД( КОДОБОРУДОВАНИЯ );
IF ( ПУСТО(АдресТСД) )
{
	СООБЩЕНИЕ("Не определить адрес ТСД. Проверьте правильность указания адреса в карточке оборудования в строке соединения и сохраните карточку торгового оборудования.", RESTAPI.ИМЯОБОРУДОВАНИЯ() );
	RETURN "";
}

ТокенМаркСкан = RESTAPI.МАРКСКАН_Авторизация( АдресТСД );

IF ( ПУСТО(ТокенМаркСкан) )
{
	СООБЩЕНИЕ("Не удалось получить токен авторизации", RESTAPI.ИМЯОБОРУДОВАНИЯ() );
	RETURN "";
}


//формируем строку для аутентификации	
Заголовки[0] = "Content-type: application/json; charset=utf-8";
ТелоЗапроса = "";
АдресРесурса = "/v3/information?token=" + ТокенМаркСкан;

try
{
	Ответ		= RESTAPI.МАРКСКАН_GETЗАПРОС( АдресТСД, АдресРесурса, ТелоЗапроса, ТокенМаркСкан, "Заголовки" );	
	IF ( !RESTAPI.МАРКСКАН_ПРОВЕРКАНАОШИБКИ( @Ответ ) ) THROW ( Ответ );	
}
catch ( ТекстСообщения )
{		
	СООБЩЕНИЕ( ТекстСообщения );
	RETURN "";
}

RETURN Ответ;
