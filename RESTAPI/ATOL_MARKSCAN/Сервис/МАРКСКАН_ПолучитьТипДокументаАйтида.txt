ТипДокумента = "";
//определим тип документа в Айтиде по названию типа документа в МаркСкан
IF ( ТИПДОКУМЕНТАТСД == "accept" ) //Поступление
	ТипДокумента = "Приёмка";
ELSE IF ( ТИПДОКУМЕНТАТСД == "shipment" ) //Расходная накладная
	ТипДокумента = "Отгрузка";
ELSE IF ( ТИПДОКУМЕНТАТСД == "revaluation" ) //Переоценка
	ТипДокумента = "Переоценка";
ELSE IF ( ТИПДОКУМЕНТАТСД == "stocktaking") //Инвентаризация ТМЦ и Инвентаризация ЕГАИС
	ТипДокумента = "Инвентаризация";
ELSE IF ( ТИПДОКУМЕНТАТСД == "transfer" ) //Внутреннее перемещение
	ТипДокумента = "Перемещение";
ELSE IF ( ТИПДОКУМЕНТАТСД == "return" ) //Возврат поставщику
	ТипДокумента = "Возврат";
ELSE IF ( ТИПДОКУМЕНТАТСД == "barcodes" ) //Сбор штрихкодов
	ТипДокумента = "Сбор штрихкодов";	

RETURN UPPER(ТипДокумента);
