Результат		= true;
IF ( _ERRORCODE !=  0 ) 
{
	Ответ		= _ERRORDESCRIPTION + CHR( 13 ) + Ответ;
	Результат	= false;
}
ELSE IF ( _RESPONSECODE >= 300 ) 
{
	IF ( !ПУСТО( СиМ ) ) 
		Ответ	= СТРОКАИЗМАССИВА( Ответ );

	ЗАГРУЗИТЬJSON( "Ошибка", ПЕРЕКОДИРОВАТЬ( Ответ, "UTF-8", "ANSI" ) );
	Ответ		= "Ошибка: " + ЗНАЧЕНИЕПОЛЯ( "Ошибка", "Details", "" ) + CHR( 13 ) + ЗНАЧЕНИЕПОЛЯ( "Ошибка", "CommonDescription", "" );
	Результат	= ЗНАЧЕНИЕПОЛЯ( "Ошибка", "ApiErrorCode" ) == "2301";
}
RETURN Результат;
