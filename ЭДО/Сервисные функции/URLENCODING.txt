// Преобразование входящей строки в URL EncodingРезультат		= "";Длина			= ДЛИНА( Строка );FOR ( Индекс = 0; Индекс < Длина; Индекс ++ ){	Символ		= ЭЛЕМЕНТ( Строка, Индекс );	IF ( ( Символ >= 48 && Символ <= 57 ) || ( Символ >= 65 && Символ <= 90 ) || ( Символ >= 97 && Символ <= 122 ) )		Результат	+= CHR( Символ );	ELSE		Результат	+= "%" + STR( Символ, 2, 0, true, false, 16 );}RETURN Результат;
