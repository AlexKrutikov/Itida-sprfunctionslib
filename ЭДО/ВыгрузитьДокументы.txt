
ПутьКФайлам	= "";
IF ( ТИП( "_ЭДОПУТЬКФАЙЛАМ" ) == "C" )
	ПутьКФайлам	= _ЭДОПУТЬКФАЙЛАМ;

ДОБАВИТЬКОНТЕКСТ( "SELECT code, ic FROM " + ИМЯТАБЛИЦЫ , "СписокДокументов" );
WHILE ( !КОНЕЦКОНТЕКСТА( "СписокДокументов" ) )
{
	IF ( СписокДокументов.code == "002" )
		ЭДО.ВыгрузкаРасходнойНакладнойВДиадок( СписокДокументов.ic, ПутьКФайлам );
	ELSE IF ( СписокДокументов.code == "022" )
		ЭДО.ВыгрузкаСчетаФактурыВДиадок( СписокДокументов.ic, ПутьКФайлам );
	ELSE IF ( СписокДокументов.code == "050" )
		ЭДО.ВыгрузкаАктаВыполненныхРаботВДиадок( СписокДокументов.ic, ПутьКФайлам );
		
	ПРОПУСТИТЬ( 1, "СписокДокументов" );
}
УДАЛИТЬКОНТЕКСТ( "СписокДокументов" );
СООБЩЕНИЕ( "Выгрузка успешно завершена." );
