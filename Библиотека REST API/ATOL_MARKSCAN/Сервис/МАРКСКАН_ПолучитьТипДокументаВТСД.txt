ТипДокумента = "";
//определим тип документа в МаркСкан по типу документа в Айтиде
IF ( ВСПИСКЕ( КОДДОКУМЕНТА, "001", "MRK" ) ) //Поступление и Маркировка - приёмка
	ТипДокумента = "accept";
ELSE IF ( КОДДОКУМЕНТА == "002" ) //Расходная накладная
	ТипДокумента = "shipment";
ELSE IF ( КОДДОКУМЕНТА == "005" ) //Переоценка
	ТипДокумента = "revaluation";
ELSE IF ( ВСПИСКЕ( КОДДОКУМЕНТА, "007", "045" ) ) //Инвентаризация ТМЦ и Инвентаризация ЕГАИС
	ТипДокумента = "stocktaking";
ELSE IF ( КОДДОКУМЕНТА == "019" ) //Внутреннее перемещение
	ТипДокумента = "transfer";
ELSE IF ( КОДДОКУМЕНТА == "055" ) //Возврат поставщику
	ТипДокумента = "return";
ELSE IF ( КОДДОКУМЕНТА == "BARCODE" ) //Сбор штрихкодов
	ТипДокумента = "barcodes";	

RETURN ТипДокумента;
