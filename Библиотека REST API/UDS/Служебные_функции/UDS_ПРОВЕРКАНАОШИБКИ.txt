Результат		= true;
IF ( _ERRORCODE !=  0 ) 
{
	Ответ		= _ERRORDESCRIPTION + CHR( 13 ) + Ответ;
	Результат	= false;
}
ELSE IF ( AT( Ответ, "errorCode" ) > 0 ) 
{
	ЗАГРУЗИТЬJSON( "Ошибка", ПЕРЕКОДИРОВАТЬ( Ответ, "UTF-8", "ANSI" ) );
	Ответ		= "Ошибка: " + ЗНАЧЕНИЕПОЛЯ( "Ошибка", "errorCode" ) + " - " + ЗНАЧЕНИЕПОЛЯ( "Ошибка", "message" );
	Результат	= false;
}
RETURN Результат;
