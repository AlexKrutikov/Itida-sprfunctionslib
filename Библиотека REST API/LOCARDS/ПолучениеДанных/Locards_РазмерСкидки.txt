//получим информацию по карте, но без вывода сообщения, на случай, если карта не скидочная, а бонусная
ИнформацияОКарте = RESTAPI.Locards_ПолучениеИнформацииОКарте( НОМЕРКАРТЫ, НОМЕРТЕЛЕФОНА, false );

IF ( ПУСТО( ИнформацияОКарте ) ) RETURN 0;

//проверим, скидочная карта или нет
Кэшбэк = VAL( ПОЛЕ_JSON (ИнформацияОКарте, "cashback", 0 ) );

IF ( Кэшбэк == 0  AND ВЫВОДИТЬСООБЩЕНИЕ )
	RESTAPI.LOCARDS_ВывестиСообщениеОКарте( ИнформацияОКарте );

РазмерСкидки = VAL( ПОЛЕ_JSON (ИнформацияОКарте, "discount", 0 ) );

RETURN РазмерСкидки;
