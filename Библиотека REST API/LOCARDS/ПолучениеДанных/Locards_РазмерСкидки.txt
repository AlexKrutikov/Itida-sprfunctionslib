ИнформацияОКарте = RESTAPI.Locards_ПолучениеИнформацииОКарте( НОМЕРКАРТЫ, НОМЕРТЕЛЕФОНА );

	RESTAPI.LOCARDS_ВывестиСообщениеОКарте( ИнформацияОКарте );
