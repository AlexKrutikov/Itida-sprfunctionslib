Результат		= true;

IF ( AT( Ответ, "error" ) > 0 ) 
{
	ОтветУспех = ПОЛЕ_JSON( Ответ, "success", false);
	ОтветОшибка = ПОЛЕ_JSON( Ответ, "error", "");
	
	IF ( !ПУСТО( ОтветОшибка ) )
	{
		КодОшибки = ПОЛЕ_JSON( Ответ, "statusCode", "");
		ТекстОшибки = ПОЛЕ_JSON( Ответ, "message", "");
		Ответ		= "Ошибка: " + КодОшибки + CHR(13) + CHR(10) + "Описание: " + ТекстОшибки;
		
		Результат	= false;
	}
}

RETURN Результат;
