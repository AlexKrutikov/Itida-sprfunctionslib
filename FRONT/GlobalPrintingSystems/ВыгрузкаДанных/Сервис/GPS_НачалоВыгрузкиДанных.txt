ДескрипторФайла			= -1;
IF ( FileExists( path_out ) AND ПУСТО( exchangeservice_out ) )
{
	ДескрипторФайла		= ФайлОткрыть( path_out, 0, 1 );
	IF ( ДескрипторФайла == -1 )
	{
		_КОДОШИБКИ		= "001";
		_ТЕКСТОШИБКИ	= "ККМ '" + ALLTRIM( name ) + "'. Не удалось открыть файл для выгрузки данных '" + ALLTRIM( path_out ) + "'.";
		_ОШИБКАВЫПОЛНЕНИЯ	= true;
		RETURN false;
	}
	// Первую строку пропускаем
	ФайлПрочитать( ДескрипторФайла, "S" );
	// Во второй строке первый символ указывает на то был ли файл обработан ККМ
	ПроверкаФлага 		= ФайлПрочитать( ДескрипторФайла, "S" );
}
