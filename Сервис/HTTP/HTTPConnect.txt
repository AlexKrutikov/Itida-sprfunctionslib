// Если не указаны параметры, то используем сохраненные ранее значенияIF ( ПУСТО( АдресСервера ) )	RETURN false;HTTPHostIP		= АдресСервера;ПозицияДТ		= AT( АдресСервера, ":" );	IF ( ПозицияДТ > 0 ){	HTTPHostIP	= ПОДСТРОКА( АдресСервера, 1, ПозицияДТ - 1 );	Порт		= ПОДСТРОКА( АдресСервера, ПозицияДТ + 1 );}IF ( ПУСТО( Порт ) )	Порт	= 80;Порт			= VAL( Порт );// Сохраняем на будущее поле загоовка HostHTTPPort		= Порт;HTTPHost		= HTTPHostIP + ":" + HTTPPort;HTTPConnection	= СОКЕТСОЕДИНИТЬ( HTTPHostIP, HTTPPort, ВестиЖурнал );RETURN HTTPConnection >= 0;
