IF (TYPE("ОТКЛЮЧАТЬККТ") == "U")
	ОТКЛЮЧАТЬККТ = true;
	
Результат		= true;
fptr = _МАССИВККТ[ VAL( ФР.КОДОБОРУДОВАНИЯ() ) ];
IF ( fptr.errorCode( ) <> 0 )
{
	// Вывод ошибки
	СООБЩЕНИЕ( "Ошибка " + fptr.errorCode( ) + " (" + МестоВызова + ")" + CHR( 13 ) + "Описание ошибки: " + fptr.errorDescription( ), "Ошибка работы с ККМ" );
	Результат	= false;
	fptr.resetError( );
}

IF (ОТКЛЮЧАТЬККТ)
	ФР.АТОЛДТО10_ОтключитьККТ();
	
RETURN Результат;
