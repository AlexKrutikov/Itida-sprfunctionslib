
Результат		= true;
fptr = _МАССИВККТ[ VAL( ФР.КОДОБОРУДОВАНИЯ() ) ];
IF ( fptr.errorCode( ) <> 0 )
{
	// Вывод ошибки
	СООБЩЕНИЕ( "Ошибка " + fptr.errorCode( ) + " (" + МестоВызова + ")" + CHR( 13 ) + "Описание ошибки: " + fptr.errorDescription( ), "Ошибка работы с ККМ" );
	Результат	= false;
	fptr.resetError( );
}

// Если нужно отсоединяться от ФР, то закрываем соединение
IF ( fptr.isOpened( ) AND ОтсоединятьсяОтУстройства )
{
	fptr.close( );
	_МАССИВККТ[ VAL( ФР.КОДОБОРУДОВАНИЯ() ) ] = fptr;
}

RETURN Результат;
