//Тип чека = 0 Продажа (приход) - В ККМ = 0
//Тип чека = 1 Возврат (возврат прихода) - В ККМ = 2
//Тип чека = 6 Прочее/Вскрытие тары (приход) (или нефискальный)
//Тип чека = 4 Расход по ФН (Расход) - В ККМ = 1
//Тип чека = 5 Возврат расхода по ФН (Возврат расхода) - В ККМ = 3

//ТипЧекаККМ	= ЕСЛИ( ТипЧека == 1, "2", ЕСЛИ( ТипЧека == 4, "1", ЕСЛИ( ТипЧека == 5, "3", "0" ) ) );

IF ( ТипЧекаАйтида == 0 ) //"Продажа"
	КодВидаЧека = 0;
ELSE IF ( ТипЧекаАйтида == 1 ) //"Возврат"
	КодВидаЧека = 2;
ELSE IF ( ТипЧекаАйтида == 4 ) //"Расход"
	КодВидаЧека = 1;
ELSE IF ( ТипЧекаАйтида == 5 ) //"Возврат расхода"
	КодВидаЧека = 3;
ELSE IF ( ТипЧекаАйтида == 6 ) //"Прочее/Вскрытие тары"
	КодВидаЧека = -1; //в ККМ такого номера нет, будем использовать как признак нефискального чека
ELSE
	КодВидаЧека = 0; //по умолчанию возвращаем Продажу (приход)
	
RETURN КодВидаЧека;
