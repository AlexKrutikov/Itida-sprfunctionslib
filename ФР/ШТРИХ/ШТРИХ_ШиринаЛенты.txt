// Ширина ленты
// Ширина стандартная - 32 символа
ШиринаЛенты = 32;

IF ( ПЕРЕМЕННАЯ("_КОЛИЧЕСТВОСИМВОЛОВЧЕКОВОЙЛЕНТЫ", " ") <> " " )
	RETURN _КОЛИЧЕСТВОСИМВОЛОВЧЕКОВОЙЛЕНТЫ;
	
IF ( !ФР.ШТРИХ_ОткрытьФР( КОДОБОРУДОВАНИЯ, "Ширина ленты в ФР" ) ) RETURN ШиринаЛенты;

driverKKM.GetDeviceMetrics();
IF ( !ФР.ШТРИХ_ЗакрытьФРСПроверкойНаОшибки( "Ширина ленты в ФР" ) ) RETURN ШиринаЛенты;

МодельККМ = VAL(driverKKM.UModel);
IF ( МодельККМ == 0 OR МодельККМ == 1 OR МодельККМ == 4 OR МодельККМ == 8 )
{
	//Штрих-ФР-Ф или Штрих-ФР-К
	ШиринаЛенты = 36;
}
ELSE IF ( МодельККМ == 2 )
{
	//Элвес-Мини-ФР-Ф
	ШиринаЛенты = 24;	
}
ELSE IF ( МодельККМ == 5 OR МодельККМ == 11 )
{
	//Штрих-950К
	ШиринаЛенты = 40;	
}
ELSE IF ( МодельККМ == 6 )
{
	//Элвес-ФР-К    
	ШиринаЛенты = 32;	
}
ELSE IF ( МодельККМ == 7 OR МодельККМ == 14 )
{
	//Штрих-Мини-ФР-К
	ШиринаЛенты = 36;
}
ELSE IF ( МодельККМ == 9 OR МодельККМ == 12 )
{
	//Штрих-Комбо-ФРК
	ШиринаЛенты = 48;
}

_КОЛИЧЕСТВОСИМВОЛОВЧЕКОВОЙЛЕНТЫ = ШиринаЛенты;			

RETURN ШиринаЛенты;
