// Номер чека в ФР
//IF ( ПЕРЕМЕННАЯ("НомерЧекаККМ", 0) <> 0 )
//	RETURN НомерЧекаККМ;
	
НомерЧекаККМ = 0;
IF ( !ФР.ШТРИХ_ОткрытьФР( КОДОБОРУДОВАНИЯ, "Номер чека" ) ) RETURN 0;
	
Прибавка = 0;
driverKKM.GetECRStatus();	
IF (driverKKM.ECRMode == 8) //чек открыт
	Прибавка = 1;

НомерЧекаККМ = VAL( driverKKM.OpenDocumentNumber ) + Прибавка;

// Закрываем соединение с ККТ
ФР.ШТРИХ_ЗакрытьФРСПроверкойНаОшибки( КОДОБОРУДОВАНИЯ, "номер чека" );

RETURN НомерЧекаККМ;
