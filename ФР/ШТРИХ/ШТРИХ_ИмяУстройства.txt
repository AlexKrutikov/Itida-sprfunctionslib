ВерсияБиблиотеки = "v.6.2";	
// Имя устройства ФР

deviceName = "ФР ШТРИХ";

try
{
	driverKKM			= СоздатьОбъект( "Addin.DrvFR", 1 );
}
catch()
{
	driverKKM = false;
}
	
IF ( ТИП(driverKKM) == "L" )
{
	deviceName = "(" + ВерсияБиблиотеки + ") Не удалось создать объект драйвера."
}
ELSE
{
	driverKKM.LDNumber			= ИНДЕКСУСТРОЙСТВА+1;
	driverKKM.GetParamLD( );
	deviceName					= "(" + ВерсияБиблиотеки + ") дрв. " + driverKKM.DriverVersion + " - " + driverKKM.LDName;
	driverKKM					= false;
}

RETURN deviceName;
