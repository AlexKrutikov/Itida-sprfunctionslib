
// Свойства драйвера ФР.
driverKKM					= СоздатьОбъект( "Addin.DrvFR", 1 );

ВерсияДрайвера = ALLTRIM(driverKKM.DriverVersion);
СовместимаяВерсия = "4.15";
i = 0;
ДрайверСовместим = true;
FOR( i=1; i <= MIN( LEN(ВерсияДрайвера), LEN(СовместимаяВерсия) ); i++)
{
	IF ( SUBSTR(ВерсияДрайвера,i,1) <> SUBSTR(СовместимаяВерсия,i,1) )
	{
		IF (SUBSTR(ВерсияДрайвера,i,1) == ".")
			CONTINUE;
		IF ( VAL( SUBSTR( ВерсияДрайвера,i,1) ) < VAL( SUBSTR( СовместимаяВерсия,i,1) ) )
		{
			ДрайверСовместим = false;
			BREAK;
		}
	}	
}
IF (!ДрайверСовместим)
	СООБЩЕНИЕ("Для передачи тэгов маркировки, а так же наименования и ИНН покупателя в составе реквизитов кассового чека необходимо использовать драйвер ККМ не ниже чем " + СовместимаяВерсия,"Версия драйвера");


driverKKM.ShowProperties( );
driverKKM					= false;
