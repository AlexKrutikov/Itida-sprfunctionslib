
IF ( TYPE( "_ФРМЕРКУРИЙ_АДРЕССЕРВЕРА" + ALLTRIM( ФР.КОДОБОРУДОВАНИЯ() ) ) == "U" )
{
	IF ( TYPE( "_ФРМЕРКУРИЙ_АДРЕССЕРВЕРА" ) == "U" )
	{
		АДРЕССЕРВЕРАФР	= "127.0.0.1:50010";
	}
	ELSE
	{
		АДРЕССЕРВЕРАФР	= _ФРМЕРКУРИЙ_АДРЕССЕРВЕРА; 
	}
}
ELSE
{
	АДРЕССЕРВЕРАФР = ВЫЧИСЛИТЬ( "_ФРМЕРКУРИЙ_АДРЕССЕРВЕРА" + ALLTRIM( ФР.КОДОБОРУДОВАНИЯ() ) );
}

//если в адрес сервера не прописали порт через двоеточие, то считаем что пор вовсе не указан и добавляем его
IF ( ATC( АДРЕССЕРВЕРАФР, ":" ) == 0 )
	АДРЕССЕРВЕРАФР += ":50010";
	
RETURN АДРЕССЕРВЕРАФР;
