//Тип чека = 0 Продажа (приход) - В ККМ = 1
//Тип чека = 1 Возврат (возврат прихода) - В ККМ = 2
//Тип чека = 6 Прочее/Вскрытие тары (приход) (или нефискальный)
//Тип чека = 4 Расход по ФН (Расход) - В ККМ = 4
//Тип чека = 5 Возврат расхода по ФН (Возврат расхода) - В ККМ = 5

//ТипЧекаККМ	= ЕСЛИ( ТипЧека == 1, "2", ЕСЛИ( ТипЧека == 4, ЕСЛИ( ЧЕККОРРЕКЦИИ, "9", "4" ), ЕСЛИ( ТипЧека == 5, "5", ЕСЛИ( ЧЕККОРРЕКЦИИ, "7", "1" ) ) ) );
	
IF ( VAL( ТипЧекаАйтида ) == 0 ) //"Продажа"
	КодВидаЧека = ЕСЛИ( ЭтоЧекКоррекции, 4, 0);
ELSE IF ( VAL( ТипЧекаАйтида ) == 1 ) //"Возврат"
	КодВидаЧека = ЕСЛИ( ЭтоЧекКоррекции, 6, 1);
ELSE IF ( VAL( ТипЧекаАйтида ) == 4 ) //"Расход"
	КодВидаЧека = ЕСЛИ( ЭтоЧекКоррекции, 5, 2);
ELSE IF ( VAL( ТипЧекаАйтида ) == 5 ) //"Возврат расхода"
	КодВидаЧека = ЕСЛИ( ЭтоЧекКоррекции, 7, 3);
ELSE IF ( VAL( ТипЧекаАйтида ) == 6 ) //"Прочее/Вскрытие тары"
	КодВидаЧека = -1; //в ККМ такого номера нет, будем использовать как признак нефискального чека
ELSE
	КодВидаЧека = 0; //по умолчанию возвращаем Продажу (приход)
	
RETURN КодВидаЧека;
