НомерПорта = VAL(ИНДЕКСУСТРОЙСТВА + 1);
ПоловинаУстройств = ЦЕЛОЕ( ФР.МЕРКУРИЙФР_КоличествоУстройств() / 2 );
IF (НомерПорта > ПоловинаУстройств )
	КОМ_Порт	= НомерПорта - ПоловинаУстройств; 
ELSE
	КОМ_Порт	= НомерПорта;
	
ИмяПорта = "(" + ФР.ВерсияБиблиотекиФР() + ") COM" + КОМ_Порт + " " + ЕСЛИ( НомерПорта > ПоловинаУстройств, "Меркурий-119Ф", "Меркурий 115Ф, 130Ф, 180Ф, 185Ф" );

RETURN ИмяПорта;
