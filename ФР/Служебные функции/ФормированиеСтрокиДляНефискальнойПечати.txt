ДлинаСтрокиПечати	= LEN(СтрокаДляПечати);
ДлинаСтрокиПечатиЦ	= LEN(СтрокаДляПечатиЦ);
ДлинаСтрокиПечатиП	= LEN(СтрокаДляПечатиП);
ОтступПечатиЛ		= "";
ОтступПечатиЦ		= "";
ОтступПечатиП		= "";
IF (ПоложениеСтроки == "L")
{
	НачальнаяПозиция	= 0;
}
ELSE IF (ПоложениеСтроки == "R")
{
	НачальнаяПозиция	= _КОЛИЧЕСТВОСИМВОЛОВЧЕКОВОЙЛЕНТЫ - LEN(СтрокаДляПечати);
	ОтступПечатиЛ 		= REPLICATE(" ", НачальнаяПозиция);
}
ELSE IF ((ПоложениеСтроки == "C") OR (ПоложениеСтроки == "С") )
{
	НачальнаяПозиция	= INT((_КОЛИЧЕСТВОСИМВОЛОВЧЕКОВОЙЛЕНТЫ - LEN(СтрокаДляПечати))/2);
	ОтступПечатиЛ 		= REPLICATE(" ", НачальнаяПозиция);
}
ELSE IF (ПоложениеСтроки == "LR")
{
	НачальнаяПозиция	= 0;
	ДлинаСтрокиПробелов	= _КОЛИЧЕСТВОСИМВОЛОВЧЕКОВОЙЛЕНТЫ - ДлинаСтрокиПечати - ДлинаСтрокиПечатиП;
	ОтступПечатиП 		= REPLICATE(" ", ДлинаСтрокиПробелов);
}
ELSE IF ( (ПоложениеСтроки == "LRC") OR (ПоложениеСтроки == "LRС") )
{
	НачальнаяПозиция	= 0;
	ДлинаСтрокиПробелов	= _КОЛИЧЕСТВОСИМВОЛОВЧЕКОВОЙЛЕНТЫ - ДлинаСтрокиПечати - ДлинаСтрокиПечатиЦ - ДлинаСтрокиПечатиП;
	IF (ПозицияЦентр <> 0 AND !ПУСТО(ПозицияЦентр))
	{
		НачальнаяПозицияЦ	= ПозицияЦентр;
	}
	ELSE
	{
		НачальнаяПозицияЦ	= INT(ДлинаСтрокиПробелов/2);
	}
	НачальнаяПозицияП	= ДлинаСтрокиПробелов - НачальнаяПозицияЦ;
	ОтступПечатиЦ 		= REPLICATE(" ", НачальнаяПозицияЦ);
	ОтступПечатиП 		= REPLICATE(" ", НачальнаяПозицияП);		
}

ИтоговаяНефискальнаяСтрока = ОтступПечатиЛ + СтрокаДляПечати + ОтступПечатиЦ + СтрокаДляПечатиЦ + ОтступПечатиП + СтрокаДляПечатиП;

RETURN ИтоговаяНефискальнаяСтрока;
