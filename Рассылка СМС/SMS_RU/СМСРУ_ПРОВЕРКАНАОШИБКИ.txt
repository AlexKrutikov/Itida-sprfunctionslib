Результат		= true;
IF ( _ERRORCODE !=  0 ) 
{
	Ответ		= _ERRORDESCRIPTION + CHR( 13 ) + Ответ;
	Результат	= false;
}
ELSE IF ( AT( Ответ, "error" ) > 0 ) 
{
	ЗАГРУЗИТЬJSON( "Ошибка", ПЕРЕКОДИРОВАТЬ( Ответ, "UTF-8", "ANSI" ) );
	IF (ЗНАЧЕНИЕПОЛЯ( "Ошибка", "status" ) <> "ОК")
	{
		Ответ		= "Ошибка: " + ЗНАЧЕНИЕПОЛЯ( "Ошибка", "status_code" ) + " (подробности см. в личном кабинете SMS.RU)";
		Результат	= false;
	}
}
RETURN Результат;
