Результат		= true;
IF ( _ERRORCODE !=  0 ) 
{
	Ответ		= _ERRORDESCRIPTION + CHR( 13 ) + Ответ;
	Результат	= false;
}
ELSE  
{
	ЗАГРУЗИТЬJSON( "Ошибка", ПЕРЕКОДИРОВАТЬ( Ответ, "UTF-8", "ANSI" ) );
	IF (ЗНАЧЕНИЕПОЛЯ( "Ошибка", "success" ) <> "true")
	{
		Ответ		= "Ошибка: " + ЗНАЧЕНИЕПОЛЯ( "Ошибка", "message" );
		Результат	= false;
	}
}
RETURN Результат;
