ДатаСравненияМЕСЯЦ = МЕСЯЦ(ДатаСравнения);
ДатаСравненияДЕНЬ = ДЕНЬ(ДатаСравнения);

ТЕКУЩАЯДАТА = Дата();

ТекущаяДатаДЕНЬ = ДЕНЬ(ТЕКУЩАЯДАТА);
ТекущаяДатаМЕСЯЦ = МЕСЯЦ(ТЕКУЩАЯДАТА);
ТекущаяДатаГОД = ГОД(ТЕКУЩАЯДАТА);

ДАТАСРАВНЕНИЕТЕКСТ = ALLTRIM(ДатаСравненияДЕНЬ) + "." + ALLTRIM(ДатаСравненияМЕСЯЦ) + "." + ALLTRIM(ТекущаяДатаГОД);
ДАТАСРАВНЕНИЕДАТА = CTOD(ДАТАСРАВНЕНИЕТЕКСТ);

РазницаВДатах = DATEDIFF( ТЕКУЩАЯДАТА, ДАТАСРАВНЕНИЕДАТА, 'day');

IF ( РазницаВДатах >= -РАЗНИЦАВДНЯХНАЧАЛО AND РазницаВДатах <= РАЗНИЦАВДНЯХОКОНЧАНИЕ )
{
	СПИСОКДАТРОЖДЕНИЯ += DTOC( ДАТАСРАВНЕНИЕДАТА, 4 ) + CHR(13) + CHR(10);
	RETURN true;
}
ELSE
	RETURN false;
