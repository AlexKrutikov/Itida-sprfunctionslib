Результат		= true;
IF ( _ERRORCODE !=  0 ) 
{
	Ответ		= _ERRORDESCRIPTION + CHR( 13 ) + Ответ;
	Результат	= false;
}
ELSE IF ( AT( Ответ, "Error" ) > 0 ) 
{
	ЗАГРУЗИТЬJSON( "Ошибка", ПЕРЕКОДИРОВАТЬ( Ответ, "UTF-8", "ANSI" ) );
	Ответ		= "Ошибка отправки смс"; //+ ЗНАЧЕНИЕПОЛЯ( "Ошибка", "error" );
	Результат	= false;
}
RETURN Результат;
